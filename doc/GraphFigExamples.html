drawEmbed :: Tgraph -> Tgraph -> (Vertex, Vertex) -> Diagram B #

drawEmbed (a,b) g1 g2 embeds g1 (coloured red) onto g2, by aligning with (a,b) vertices a and b must be common to g1 and g2

drawEmbedVP :: VPatch -> VPatch -> (Vertex, Vertex) -> Diagram B #
